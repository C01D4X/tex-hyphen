%% Conversion from UTF-8 to IL3 for 8-bit TeX engines

\catcode"C4=\active
\catcode"C5=\active

\def^^c4#1{%
	\ifx#1^^89^^e6\else % ĉ - ccircumflex
	\ifx#1^^9d^^f8\else % ĝ - gcircumflex
	\ifx#1^^a5^^b6\else % ĥ - hcircumflex
	\ifx#1^^b5^^bc\else % ĵ - jcircumflex
	\errmessage{Hyphenation pattern file corrupted!}
\fi\fi\fi\fi}
\def^^c5#1{%
	\ifx#1^^9d^^fe\else % ŝ - scircumflex
	\ifx#1^^ad^^fd\else % ŭ - ubreve
	\errmessage{Hyphenation pattern file corrupted!}
\fi\fi}

% ensure all the chars above have valid lccode values
\lccode"B6="B6 % ĥ - hcircumflex
\lccode"BC="BC % ĵ - jcircumflex
\lccode"E6="E6 % ĉ - ccircumflex
\lccode"F8="F8 % ĝ - gcircumflex
\lccode"FD="FD % ŭ - ubreve
\lccode"FE="FE % ŝ - scircumflex
