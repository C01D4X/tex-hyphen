	\ifx#1^^b1^^b4\else % i
	\errmessage{Hyphenation pattern file corrupted!}
\fi}

\fi\fi\fi\fi\fi\fi\fi\fi\fi\fi\fi\fi\fi\fi\fi\fi}

\fi}
